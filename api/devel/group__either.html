template<typename R , typename _L , typename L = typename std::decay<_L>::type>
constexpr either<L,R> ftl::make_left ( _L &&  l)
noexcept

Smart constructor of left values.

Note that L can be deduced by the compiler by the value passed to make_left, hence you need only provide the R template parameter.

example:

either<int,float> e = make_left<float>(12);
